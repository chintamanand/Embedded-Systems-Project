#include<LPC17xx.h>

unsigned int i,j;
unsigned long LED =0X00000010;

int main(void){
	SystemInit();
	SystemCoreClockUpdate();
	LPC_PINCON->PINSEL0 &= 0xFF0000FF;
	
	LPC_GPIO0->FIODIR |= 0X00000FF0;
	while(1)
	{
	unsigned int a=0;
		LED=0x00000010;
		for(a=0;a<256;a++){
			LPC_GPIO0->FIOSET=a<<4;
		
			
			for(j=0;j<10000;j++);
			LPC_GPIO0->FIOCLR=0xFFFF;
		
		}
		if (a==256)
		a=0;}}
